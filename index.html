TemplateCraft

Inference-based template generator for Go Kart.

Files

  • template-craft.js
  • template-craft.min.js

Usage

<script src="dist/template-craft.min.js"></script>
<script>
  const templates = templateCraft.generate(schema, {
    collectionName: 'client'
  });
  const pages = templateCraft.generatePages(schema, {
    collectionName: 'client'
  });
</script>
